/**
 * Itera sobre a aba lotacao, tenta atualizar o link para a aba com o nome do prestador na planilha entidades
 * atraves da funcao atualizarLinkPrestador. 
 */
function verificarReferenciaDosPrestadores() {
  let colunaPrestadores = variaveis.abaLotacao.getRange(2, variaveis.COLUNA_PRESTADORES, variaveis.abaLotacao.getLastRow());
  let colunaEntidades = variaveis.abaLotacao.getRange(2, variaveis.COLUNA_ENTIDADES_LOTACAO, variaveis.abaLotacao.getLastRow());
  for (let i = 1; i <= colunaPrestadores.getHeight(); i++) {
    let entidade = colunaEntidades.getCell(i, 1).getValue();
    let linkEntidade = procurarLinkEntidade(entidade);
    let prestador = colunaPrestadores.getCell(i, 1);
    if (linkEntidade != "" && !prestador.getRichTextValue().getLinkUrl()) {
      atualizarLinkDoPrestador(linkEntidade, prestador);
    }
  }
}

/**
 * Verifica se existe uma aba com o mesmo nome contido na celula 'prestador recebida como parametro
 * e atualiza a propria celula com um texto com link imbutido, o link leva diretamente
 * para a aba do prestador na planilha entidade.
 * @param {String} link - link da planilha de entidade 
 * @param {GoogleAppsScript.Spreadsheet.Range} prestador - celula que possui o nome do prestador.
 */
function atualizarLinkDoPrestador(link, prestador){
  try{
    console.log("Atualizando link do prestador: " + prestador.getValue());
    let planilhaEntidade = SpreadsheetApp.openByUrl(link);
    let abaPrestador = planilhaEntidade.getSheetByName(prestador.getValue());
    if(abaPrestador == null){
      throw "Não foi encontrado relatório do prestador, necessário verificar manualmente.";
    }
    let idPlanilhaPrestador = abaPrestador.getSheetId();
    let conteudoCelula = criarTextoComLink(prestador.getValue(), `${planilhaEntidade.getUrl()}#gid=${idPlanilhaPrestador}`);
    prestador.setRichTextValue(conteudoCelula);
  } catch(erro){
    variaveis.planilha.toast(erro, prestador.getValue(), 10);
  }
}

/**
 * Utiliza a propriedade newRichTextValue para criar um texto clicavel na planilha.
 * @param {String} texto 
 * @param {String} link 
 * @returns {GoogleAppsScript.Spreadsheet.RichTextValue} Texto com o link aplicado.
 */
function criarTextoComLink(texto, link){
  let construtorTextoComLink = SpreadsheetApp.newRichTextValue();
  construtorTextoComLink = construtorTextoComLink.setText(texto);
  construtorTextoComLink = construtorTextoComLink.setLinkUrl(link);
  let textoComLink = construtorTextoComLink.build();
  return textoComLink;
}
