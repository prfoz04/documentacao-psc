/**
 * atualização de entidade pelo site, é um acionador que procura algo preenchido na coluna temporario e 
 * ao encontrar realiza a função de alterar entidade  ou remover psc
 */
function alterarEntidadeSite() {
  
  let colunaTemporario = variaveis.abaLotacao.getRange("U:U").getValues();
  colunaTemporario = colunaTemporario.map(value => value[0]);

  for (let i = 1; i < colunaTemporario.length; i++) {

    if (colunaTemporario[i] != '') {
      let prestador = variaveis.abaLotacao.getRange(i + 1, variaveis.COLUNA_PRESTADORES).getValue();
      let celula = variaveis.abaLotacao.getRange(i + 1, variaveis.COLUNA_ENTIDADES_LOTACAO);

      if (colunaTemporario[i] == 'encerrado') {
        enviaEmails();
        console.log(`Encerrando PSC para o prestador ${prestador}`);
        tratarPenaConcluida(celula, celula.getValue());
      }

      else {
        let novaEntidade = celula.getValue();
        let entidadeAnterior = colunaTemporario[i];
        console.log(`Alterando entidade para o prestador ${prestador}, de '${entidadeAnterior}' para '${novaEntidade}'`);
        editarEntidadePrestador(celula, entidadeAnterior);
      }
      variaveis.abaLotacao.getRange(i + 1, 21).setValue('');
    }
  } 
}

/**
 * Verifica se existe uma nova linha na coluna loacao, caso exista, cria o registro do prestador na entidade 
 * e envia um email automatico de aviso a entidade.
 * Possui um acionador de tempo.
 */
function verificarNovosPrestadores() { 

  let linha = variaveis.abaLotacao.getLastRow();
  let linhasProcessadas = 0;

  while (linha > 0) {    

    const valores = variaveis.abaLotacao.getRange(linha, 1, 1, 11).getValues()[0]; // Colunas A até H
    const [colA, colB, colC, colD, colE, , , colH, , , colK] = valores;
    const valorK = colK?.toString().trim();
    const isKvaziaOuErro = !valorK || valorK.startsWith("#");

    // Se todos os campos A–E e H estiverem preenchidos E K estiver válido → já foi processado → parar o loop
    if (colA && colB && colC && colD && colE && colH && !isKvaziaOuErro) {
      Logger.log(`Parando na linha ${linha}, dados já processados.`);
      break;
    }
    // Se os campos A–E e H estiverem preenchidos e k vazio/erro, então processa
    if (colA && colB && colC && colD && colE && colH && isKvaziaOuErro) {
      try {        
        const celulaEntidade = variaveis.abaLotacao.getRange(linha, 8); // Coluna H        
        adicionarNovoPrestador(celulaEntidade);
        const nomePrestador = colB;                      // Nome do prestador (coluna B)
        const entidadeNome = colH;                       // Nome da entidade (coluna H)        
        const max_horas = variaveis.abaLotacao.getRange(linha, 5).getDisplayValue(); // Coluna E
        const classe = variaveis.abaLotacao.getRange(linha, 15).getValue();          // Coluna O (15)
        enviaEmailEntidade(nomePrestador, entidadeNome, max_horas, classe);
        Logger.log(`Prestador adicionado da linha ${linha}`);
        Utilities.sleep(2000); // pausa de 2 segundos
        linhasProcessadas++;

      } catch (e) {
        Logger.log(`Erro na linha ${linha}: ${e}`);
      }
    } else {
      Logger.log(`Linha ${linha} ignorada: dados incompletos (A-E e H devem estar preenchidos)`);
    }

    linha--; // Sobe uma linha
  }

  Logger.log(`Total de linhas processadas: ${linhasProcessadas}`);
}
