/**  
 * Ao editar um dado, a funcao eh disparada e verifica se deve alterar nas outras abas o dado correspondente 
 * ou adicionar um novo prestador, importante para a consistencia dos dados entre as abas.
 * Disparado por um acionador de edicao.
 * @param {*} evento 
 */
function acionadorEdicao(evento) {
  let celula = evento.range;
  let valorAnterior = evento.oldValue;

  if (variaveis.planilha.getActiveSheet().getName() == variaveis.abaLotacao.getName()) {
    console.log("Aba ativa Lotação");
    if (celula.getColumn() == variaveis.COLUNA_ENTIDADES_LOTACAO) {
      if (valorAnterior == null) {   //NÃO TINHA ENTIDADE
        console.log("Adicionar novo prestador");
        verificarNovosPrestadores();
      }
       else {   //TINHA ENTIDADE
        console.log("Editar entidade prestador");
        editarEntidadePrestador(celula, valorAnterior);

      }
    }
  }
}

/**
 * Itera sobre a aba 'Quadro Vagas' registrando os dados de cada entidade.
 * Função chamada na declaração da variável entidades.
 * @return {entidades[]} - array de objetos que representam entidades.
 */
function pegarEntidades() {
  let entidadesDados = [];
  let abaEntidadesDados = variaveis.abaQuadroVagas.getRange(3, 1, variaveis.abaQuadroVagas.getLastRow(), 5);
  for (let i = 1; i < abaEntidadesDados.getHeight(); i++) {
    let entidade = {
      email: "",
      nome: "",
      link: "",
      endereco: ""
    };
    if (abaEntidadesDados.getCell(i, 4).getValue() != "" || abaEntidadesDados.getCell(i, 5).getValue() != "") {
      entidade.email = abaEntidadesDados.getCell(i, 1).getValue();
      entidade.nome = abaEntidadesDados.getCell(i, 3).getValue();
      entidade.link = abaEntidadesDados.getCell(i, 4).getValue();
      entidade.endereco = abaEntidadesDados.getCell(i, 5).getValue();
      entidadesDados.push(entidade);
    }
  }
  return entidadesDados;
}

/**
 * Concede permissao de editor em todas as planilhas de entidade.
 * @param {string} email 
 */
function darPermissao(email) {
  for (let entidade of entidades) {
    if (entidade.nome != "xPena concluída") {
      let planilhaEntidade = SpreadsheetApp.openByUrl(entidade.link);
      for (let aba of planilhaEntidade.getSheets()) {
        aba.protect().addEditor(email);
      }
      Logger.log(`Concedido para ${email} permissão de editor na planilha ${entidade.nome}`)
    }
  }
}
